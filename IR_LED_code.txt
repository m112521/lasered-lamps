#include <IRremote.hpp>
#include <Adafruit_NeoPixel.h>

#define IR_RECEIVE_PIN 2
#define IR_BUTTON_PLUS 21
#define IR_BUTTON_MINUS 7

#define MATRIX_PIN 8
#define LED_COUNT 16

Adafruit_NeoPixel matrix = Adafruit_NeoPixel(LED_COUNT, MATRIX_PIN, NEO_GRB + NEO_KHZ800);


void setup(){
  Serial.begin(9600);
  IrReceiver.begin(IR_RECEIVE_PIN);

  matrix.begin();
}

void loop(){
   if (IrReceiver.decode()) {
      IrReceiver.resume(); // Enable receiving of the next value
      int command = IrReceiver.decodedIRData.command;
      Serial.println(command);
      switch (command) {
        case IR_BUTTON_PLUS: {
          colorWipe(matrix.Color(255, 255, 120), 50);
          break;
        }
        case IR_BUTTON_MINUS: {
          crossBlink(matrix.Color(255, 0, 0), 50);
          break;
        }
      }
  }
}

void crossBlink(uint32_t c, uint8_t wait) {
  for (uint16_t i = 0; i < matrix.numPixels(); i++) {
    if (i % 5 == 0 || i % 3 == 0) { 
      matrix.setPixelColor(i, c);
      //delay(wait);
    }
    else {
      matrix.setPixelColor(i, matrix.Color(0, 0, 0));
    }
    matrix.show();
  }
}

void colorWipe(uint32_t c, uint8_t wait) {
  for (uint16_t i = 0; i < matrix.numPixels(); i++) {
    matrix.setPixelColor(i, c);
    matrix.show();
  }
  //delay(wait);
}
