#include <IRremote.hpp>
#include <Adafruit_NeoPixel.h>

#define IR_RECEIVE_PIN 2
#define IR_BUTTON_PLUS 21
#define IR_BUTTON_MINUS 7
#define IR_EQ 9
#define IR_PREV 68
#define IR_NEXT 64

#define MATRIX_PIN 8
#define LED_COUNT 16

Adafruit_NeoPixel matrix = Adafruit_NeoPixel(LED_COUNT, MATRIX_PIN, NEO_GRB + NEO_KHZ800);

int currentState = 0; // 0 - off; 1 - disko; 2 - white;

void setup(){
  Serial.begin(9600);
  IrReceiver.begin(IR_RECEIVE_PIN);

  matrix.begin();
}

void loop(){
  if (IrReceiver.decode()) {
      IrReceiver.resume(); // Enable receiving of the next value
      int command = IrReceiver.decodedIRData.command;
      Serial.println(command);
      switch (command) {
        case IR_EQ: {
          currentState = 2;
          break;
        }
        case IR_NEXT: {
          currentState = 0;
          break;
        }
        case IR_PREV: {
          currentState = 1;
          break;
        }
      }
      switch (currentState) {
        case 2:
          colorWipe(matrix.Color(random(0,255), random(0,255), random(0,255)), 100);
          break;
        case 0:
          colorWipe(matrix.Color(0, 0, 0), 0);
          break;
        case 1:
          colorWipe(matrix.Color(255, 255, 255), 0);
          break;
      }

  }
}

void colorWipe(uint32_t c, uint8_t wait) {
  for (uint16_t i = 0; i < matrix.numPixels(); i++) {
    matrix.setPixelColor(i, c);
    matrix.show();
  }
  delay(wait);
}
