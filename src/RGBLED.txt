#include <IRremote.hpp>

#define IR_RECEIVE_PIN 2
#define IR_BUTTON_PLUS 21
#define IR_BUTTON_MINUS 7
#define IR_EQ 9
#define IR_PREV 68
#define IR_NEXT 64

 
int currentState = 0; // 0 - off; 1 - disko; 2 - white;

void setup(){
  Serial.begin(9600);
  IrReceiver.begin(IR_RECEIVE_PIN);
  pinMode(A0, OUTPUT);
pinMode(A1, OUTPUT);
pinMode(A2, OUTPUT);  
}

void loop(){
  if (IrReceiver.decode()) {
      IrReceiver.resume(); // Enable receiving of the next value
      int command = IrReceiver.decodedIRData.command;
      Serial.println(command);
      switch (command) {
        case IR_EQ: {
          currentState = 2;
          break;
        }
        case IR_NEXT: {
          currentState = 0;
          break;
        }
        case IR_PREV: {
          currentState = 1;
          break;
        }
      }
      switch (currentState) {
        case 2:
          colorWipe(random(0,255), random(0,255), random(0,255));
          break;
        case 0:
          colorWipe(0, 0, 0);
          break;
        case 1:
          colorWipe(255, 255, 255);
          break;
      }

  }
}

void colorWipe(int r, int g, int b) {
  analogWrite(A0, r);
  analogWrite(A1, g);
  analogWrite(A2, b);
}
